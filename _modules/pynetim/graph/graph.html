[文档]    def in_neighbors(self, node):
        """
        获取指定节点的入邻居节点。

        对于有向图，返回前驱节点；对于无向图，返回所有邻居节点。

        Args:
            node: 节点标识符

        Returns:
            iterator: 指定节点的入邻居节点迭代器
        """
        return self.graph.predecessors(node) if self.direction else self.graph.neighbors(node)
